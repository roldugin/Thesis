%% Automatically make code after underscore subscripted

\def\SubscriptWord#1 {\check#1\relax\textsubscript{#1} }

% Check function parses world until space/newline
\def\check#1{%
\ifx\relax#1%
\else
\ifcat a#1%
\else
\typeout{illegal character #1}%
\fi
\expandafter\check
\fi}

  {ccode}[1]
